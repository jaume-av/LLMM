# EXERCICI 7 – Llista de videojocs

```javascript
// Creem un array amb tres videojocs
const videojocs = ["Minecraft", "FIFA", "Tetris"];

// Mostrem l'array complet
console.log(videojocs);

// Accedim als elements per índex
console.log(videojocs[0]);
console.log(videojocs[1]);
console.log(videojocs[2]);

// Modifiquem el segon element
videojocs[1] = "Mario Kart";

// Mostrem l'array modificat
console.log(videojocs);
```

# EXERCICI 8 – Compra setmanal

```javascript
// Creem un array buit
const compra = [];

// Afegim elements al final
compra.push("pa");
compra.push("llet");
compra.push("ous");

// Mostrem l'array
console.log(compra);

// Mostrem la longitud
console.log(compra.length);

// Eliminem l'últim element
compra.pop();

// Mostrem l'array després de pop
console.log(compra);

// Mostrem la nova longitud
console.log(compra.length);
```

# EXERCICI 9 – Assignatures del curs

```javascript
// Creem un array amb assignatures
const assignatures = ["Matemàtiques", "Valencià", "Història", "Anglés", "Biologia"];

// Recorrem l'array amb for
for (let i = 0; i < assignatures.length; i++) {
    console.log(assignatures[i]);
}

// Recorrem l'array amb forEach
assignatures.forEach(function(assignatura) {
    console.log(assignatura);
});

// Busquem la posició d'un element
console.log(assignatures.indexOf("Història"));

// Busquem la posició d'un element que no existeix
console.log(assignatures.indexOf("Física"));
```

# EXERCICI 10 – Ciutats favorites

```javascript
// Creem un array amb ciutats
const ciutats = ["València", "Barcelona", "Sevilla", "Bilbao", "Granada"];

// Unim tots els elements en una cadena
console.log(ciutats.join(" - "));

// Creem una còpia parcial amb slice
const subarray = ciutats.slice(1, 4);

// Mostrem el nou array
console.log(subarray);

// Mostrem l'array original
console.log(ciutats);

// Eliminem dos elements amb splice
ciutats.splice(1, 2);

// Mostrem l'array després del canvi
console.log(ciutats);
```

# EXERCICI 12 – Informació d’un mòbil

```javascript
// Creem un objecte
const mobil = {
    marca: "Samsung",
    model: "Galaxy S21",
    preu: 800
};

// Mostrem l'objecte
console.log(mobil);

// Modifiquem el preu
mobil.preu = 750;

// Afegim una nova propietat
mobil.disponible = true;

// Afegim una propietat amb espais

// Mostrem l'objecte final

```

# EXERCICI 13 – Sèrie de televisió

```javascript
// Creem un objecte amb informació d'una sèrie
const serie = {
    titol: "Dark",
    plataforma: "Netflix",
    temporades: 3,
    acabada: true
};

// Mostrem l'objecte
console.log(serie);

// Eliminem una propietat
delete serie.acabada;

// Comprovem si existeixen propietats
console.log("temporades" in serie);
console.log("director" in serie);

// Recorrem l'objecte amb for...in
for (let clau in serie) {
    console.log(clau + ": " + serie[clau]);
}
```

# EXERCICI 14 – Biblioteca digital

```javascript
// Creem un array d'objectes
const llibres = [
    { titol: "1984", genere: "Distopia", pagines: 328 },
    { titol: "El Hobbit", genere: "Fantasia", pagines: 310 },
    { titol: "Dune", genere: "Ciència-ficció", pagines: 412 }
];

// Accedim a propietats d'objectes dins de l'array
console.log(llibres[0].titol);
console.log(llibres[1].genere);
console.log(llibres[2].pagines);

// Recorrem l'array
for (let i = 0; i < llibres.length; i++) {
    console.log("Títol: " + llibres[i].titol);
    console.log("Gènere: " + llibres[i].genere);
    console.log("Pàgines: " + llibres[i].pagines);
}

// Copiem el primer objecte
const copia = Object.assign({}, llibres[0]);

// Mostrem la còpia
console.log(copia);

// Mostrem les claus
console.log(Object.keys(llibres[0]));

// Mostrem els valors
console.log(Object.values(llibres[0]));

// Congelarem la còpia
Object.freeze(copia);

// Intentem modificar una propietat
copia.pagines = 500;

// Mostrem la còpia després de l'intent de modificació
console.log(copia);
```
