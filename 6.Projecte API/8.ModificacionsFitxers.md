# Ciutats


## `routes/ciutats.js`

```javascript
router.put('/:id', function(req, res) {
    db('CIUTAT')
        .where('ID', req.params.id)
        .update({
            NOM: req.body.NOM,
            DESCRIPCIO: req.body.DESCRIPCIO,
            IMATGE: req.body.IMATGE,
            POBLACIO: req.body.POBLACIO,
            PROVINCIA_ID: req.body.PROVINCIA_ID
        })
        .then(function(resultat) {
            res.json({ missatge: 'Ciutat modificada correctament' });
        })
        .catch(function(error) {
            res.status(500).json({ error: 'Error en la base de dades' });
        });
});
```

### Vista de modificació

```javascript
function vistaModificarCiutat(id) {
    return `
        <section x-data="modificarCiutatApp(${id})"
                 x-init="carregarCiutat()">

            <h2>Modificar ciutat</h2>

            <form @submit.prevent="modificarCiutat()">

                <label for="nom">Nom</label>
                <input id="nom" type="text" x-model="ciutat.NOM">

                <label for="descripcio">Descripció</label>
                <input id="descripcio" type="text" x-model="ciutat.DESCRIPCIO">

                <label for="poblacio">Població</label>
                <input id="poblacio" type="number" x-model="ciutat.POBLACIO">

                <label for="provincia">Província</label>
                <input id="provincia" type="number" x-model="ciutat.PROVINCIA_ID">

                <label for="imatge">Imatge</label>
                <input id="imatge" type="text" x-model="ciutat.IMATGE">

                <p>
                    <button type="submit">Guardar canvis</button>

                    <button type="button"
                            class="warning"
                            onclick="mostrarCiutats()">
                        Cancel·lar
                    </button>
                </p>

            </form>

        </section>
    `;
}
```

### App Alpine

```javascript
window.modificarCiutatApp = function (id) {
    return {
        ciutat: {
            NOM: '',
            DESCRIPCIO: '',
            IMATGE: '',
            POBLACIO: '',
            PROVINCIA_ID: ''
        },

        carregarCiutat() {
            fetch('/api/ciutats/' + id)
                .then(function (resposta) {
                    return resposta.json();
                })
                .then((dades) => {
                    this.ciutat = dades;
                });
        },

        modificarCiutat() {
            fetch('/api/ciutats/' + id, {
                method: 'PUT',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify(this.ciutat)
            })
                .then(function (resposta) {
                    return resposta.json();
                })
                .then(function () {
                    mostrarCiutats();
                });
        }
    };
};
```

---

## `public/js/spa.js`

```javascript
function mostrarModificarCiutat(id) {
    document.getElementById("spa").innerHTML = vistaModificarCiutat(id);
    Alpine.initTree(document.getElementById("spa"));
}
```
