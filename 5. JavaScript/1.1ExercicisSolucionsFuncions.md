## EXERCICI 15 – Salutació bàsica

```javascript
function saludar() {
    console.log("Hola!");
}

saludar();
saludar();
```

```javascript
// Eixida:
Hola!
Hola!
```

## EXERCICI 16 – Presentació amb paràmetres

```javascript
function presentar(nom, edat) {
    console.log(nom + " té " + edat + " anys");
}

presentar("Pau", 17);
presentar("Laia", 16);
```

```javascript
// Eixida:
Pau té 17 anys
Laia té 16 anys
```

## EXERCICI 17 – Sumar i retornar

```javascript
function sumar(a, b) {
    return a + b;
}

const resultat1 = sumar(4, 6);
const resultat2 = sumar(10, 3);

console.log(resultat1);
console.log(resultat2);
```

```javascript
// Eixida:
10
13
```

## EXERCICI 18 – Return i console.log

```javascript
function mostrarDoble(num) {
    console.log(num * 2);
}

function retornarDoble(num) {
    return num * 2;
}

mostrarDoble(5);

const valor1 = mostrarDoble(5);
console.log(valor1);

const valor2 = retornarDoble(5);
console.log(valor2);
```

```javascript
// Eixida:
10
10
undefined
10
```

## EXERCICI 19 – Funcions anònimes i funcions com a valors

```javascript
const multiplicar = function(x, y) {
    return x * y;
};

console.log(multiplicar);
multiplicar(3, 4);
console.log(multiplicar(3, 4));
```

```javascript
// Eixida:
[Function]
12
```

## EXERCICI 20 – Funcions com a paràmetres i arrow functions

```javascript
function calcular(a, b, operacio) {
    return operacio(a, b);
}

const sumar = (a, b) => a + b;
const restar = (a, b) => a - b;

console.log(calcular(8, 2, sumar));
console.log(calcular(8, 2, restar));
```

```javascript
// Eixida:
10
6
```

## EXERCICI 21 – Arrow functions amb arrays

```javascript
const notes = [4, 7, 9];

notes.forEach(nota => {
    console.log(nota);
});

const aprovat = nota => {
    if (nota >= 5) {
        return "Aprovat";
    } else {
        return "Suspés";
    }
};

console.log(aprovat(4));
console.log(aprovat(7));
console.log(aprovat(9));
```

```javascript
// Eixida:
4
7
9
Suspés
Aprovat
Aprovat
```
