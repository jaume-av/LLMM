## EXERCICI 1 – Temperatura i recomanació

```javascript
// Demanem la temperatura i la convertim a número
const temperatura = Number(prompt("Introdueix una temperatura:"));

// Variable per guardar el resultat
let missatge;

// Classifiquem la temperatura
if (temperatura < -10 || temperatura > 50) {
    missatge = "Valor no habitual";
} else if (temperatura > 32) {
    missatge = "Fa molta calor";
} else if (temperatura >= 20 && temperatura <= 32) {
    missatge = "Temperatura agradable";
} else {
    missatge = "Fa fred";
}

// Mostrem el resultat en alerta i en consola
alert(missatge);
console.log(missatge);
```

## EXERCICI 2 – Usuari i contrasenya

```javascript
// Definim les dades correctes
const usuariCorrecte = "admin";
const clauCorrecta = "1234";

// Demanem les dades a l'usuari
const usuari = prompt("Introdueix el nom d'usuari:");
const clau = prompt("Introdueix la contrasenya:");

// Comprovem les dades introduïdes
if (usuari === usuariCorrecte && clau === clauCorrecta) {
    alert("Accés correcte");
} else if (usuari !== usuariCorrecte) {
    alert("Usuari incorrecte");
} else {
    alert("Contrasenya incorrecta");
}
```

## EXERCICI 3 – Classificar un número especial

```javascript
// Demanem un número enter
const numero = Number(prompt("Introdueix un número enter:"));

// Classifiquem el número
if (numero === 0) {
    alert("Zero");
} else if (numero > 0 && numero % 2 === 0) {
    alert("Positiu parell");
} else if (numero > 0 && numero % 2 !== 0) {
    alert("Positiu senar");
} else if (numero < 0 && numero % 2 === 0) {
    alert("Negatiu parell");
} else {
    alert("Negatiu senar");
}
```

## EXERCICI 4 – Els 3 bucles

```javascript
// Mostrem els múltiples de 3 del 3 al 15
for (let i = 3; i <= 15; i += 3) {
    console.log(i);
}

// Mostrem els números parells de 10 a 0
let numero = 10;

while (numero >= 0) {
    console.log(numero);
    numero -= 2;
}

// Demanem un número fins que estiga entre 1 i 10
let valor;

do {
    valor = Number(prompt("Introdueix un número entre 1 i 10:"));
} while (valor < 1 || valor > 10);

// Mostrem el número correcte
console.log("Número correcte: " + valor);
```

## EXERCICI 5 – Menú amb switch + bucle

```javascript
// Variable per guardar l'opció del menú
let opcio = "";

// El menú es repeteix fins que l'usuari confirme que vol eixir
while (opcio !== "0") {

    // Mostrem el menú
    opcio = prompt(
        "1 - Calcular el quadrat d'un número\n" +
        "2 - Dir si un número és positiu, negatiu o zero\n" +
        "3 - Comptar cap arrere des d'un número fins a 0\n" +
        "0 - Eixir"
    );

    // Gestionem les opcions
    switch (opcio) {

        case "1":
            // Demanem un número i mostrem el quadrat
            const numQuadrat = Number(prompt("Introdueix un número:"));
            console.log(numQuadrat * numQuadrat);
            break;

        case "2":
            // Demanem un número i indiquem si és positiu, negatiu o zero
            const num = Number(prompt("Introdueix un número:"));

            if (num > 0) {
                console.log("Positiu");
            } else if (num < 0) {
                console.log("Negatiu");
            } else {
                console.log("Zero");
            }
            break;

        case "3":
            // Demanem un número i fem un compte arrere fins a 0
            let inici = Number(prompt("Introdueix un número:"));

            while (inici >= 0) {
                console.log(inici);
                inici--;
            }
            break;

        case "0":
            // Confirmem si realment vol eixir
            const eixir = confirm("Segur que vols eixir?");

            if (eixir === true) {
                console.log("Programa finalitzat");
            } else {
                opcio = "";
            }
            break;

        default:
            // Opció incorrecta
            console.log("Opció incorrecta");
    }
}
```

## EXERCICI 6 – Control de mesos amb for, switch i if

```javascript
// Demanem quants mesos es volen revisar
const quantitat = Number(prompt("Quants mesos vols revisar?"));

// Repetim el procés
for (let i = 1; i <= quantitat; i++) {

    // Demanem el número de mes
    const mes = Number(prompt("Introdueix un número de mes:"));

    // Comprovem si el mes és vàlid
    if (mes < 1 || mes > 12) {
        console.log("Mes no vàlid");
    } else {

        // Classifiquem el mes segons l'estació
        switch (mes) {

            case 12:
            case 1:
            case 2:
                console.log("Hivern");
                break;

            case 3:
            case 4:
            case 5:
                console.log("Primavera");
                break;

            case 6:
            case 7:
            case 8:
                console.log("Estiu");
                break;

            case 9:
            case 10:
            case 11:
                console.log("Tardor");
                break;
        }
    }
}
```
