

### 1. Crear el nou `index.html`

Dins de la carpeta `public`, crea un fitxer nou:

```text
index.html
```

i copia este contingut:

```html
<!DOCTYPE html>
<html lang="ca">
<head>
    <meta charset="UTF-8">
    <title>Projecte API</title>

    <link rel="stylesheet"
          href="https://cdnjs.cloudflare.com/ajax/libs/picnic/7.1.0/picnic.min.css">

    <script src="js/app.js" defer></script>
    <script src="js/ciutats.js" defer></script>
    <script src="js/paisos.js" defer></script>
    <script src="js/provincies.js" defer></script>

    <script defer
            src="https://cdnjs.cloudflare.com/ajax/libs/alpinejs/3.15.0/cdn.min.js">
    </script>
</head>
<body>

<main class="container">

    <h1>Projecte API</h1>

    <nav>
        <button onclick="mostrarCiutats()">Ciutats</button>
        <button onclick="mostrarPaisos()">Països</button>
        <button onclick="mostrarProvincies()">Províncies</button>
    </nav>

    <hr>

    <div id="app"></div>

</main>

</body>
</html>
```

```text

```

```javascript
function mostrarCiutats() {
    document.getElementById("app").innerHTML = vistaCiutats();
    Alpine.initTree(document.getElementById("app"));
}

function mostrarPaisos() {
    document.getElementById("app").innerHTML = vistaPaisos();
    Alpine.initTree(document.getElementById("app"));
}

function mostrarProvincies() {
    document.getElementById("app").innerHTML = vistaProvincies();
    Alpine.initTree(document.getElementById("app"));
}

window.onload = function () {
    mostrarCiutats();
};
```



### 3. Refactoritzar `ciutats.js`


```javascript
function vistaCiutats() {
    return `
        <section x-data="ciutatsApp()" x-init="carregarCiutats()">

            <h2>Llista de ciutats</h2>

            <table class="table">
                <thead>
                    <tr>
                        <th>ID</th>
                        <th>Nom</th>
                        <th>Descripció</th>
                        <th>Població</th>
                        <th>Província</th>
                        <th>Imatge</th>
                    </tr>
                </thead>

                <tbody>
                    <template x-for="ciutat in ciutats" :key="ciutat.ID">
                        <tr>
                            <td x-text="ciutat.ID"></td>
                            <td x-text="ciutat.NOM"></td>
                            <td x-text="ciutat.DESCRIPCIO"></td>
                            <td x-text="ciutat.POBLACIO"></td>
                            <td x-text="ciutat.PROVINCIA_NOM"></td>
                            <td x-text="ciutat.IMATGE"></td>
                        </tr>
                    </template>
                </tbody>
            </table>

        </section>
    `;
}

window.ciutatsApp = function () {
    return {
        ciutats: [],

        carregarCiutats() {
            fetch('/api/ciutats')
                .then(function (resposta) {
                    return resposta.json();
                })
                .then((dades) => {
                    this.ciutats = dades;
                });
        }
    };
};
```
